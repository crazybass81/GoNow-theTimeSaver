## 📱 화면 구조 개요 / Screen Structure Overview

### 화면 목록

```
Go Now MVP - 6개 핵심 화면
├── 1. 인증 (Authentication)
│   ├── 1-1. 로그인 화면
│   └── 1-2. 회원가입 화면
│
├── 2. 개인설정 (Settings)
│   └── 2-1. 설정 화면
│
├── 3. 대시보드 (Dashboard) ⭐ 메인 화면
│   ├── 시각적 알람 (Visual Countdown)
│   ├── 선택 경로 표시 (Route Map)
│   └── 이후 스케줄 (Upcoming Schedule)
│
├── 4. 데일리 스케줄 (Daily Schedule)
│   └── 4-1. 오늘 일정 타임라인
│
├── 5. 먼슬리 스케줄 (Monthly Calendar)
│   └── 5-1. 월간 캘린더 뷰
│
└── 6. 스케줄 추가/설정 (Add/Edit Trip)
    ├── 6-1. 목적지 입력
    ├── 6-2. 시간 설정
    ├── 6-3. 준비 시간 설정
    └── 6-4. 페널티 설정
```

```dart
/**
 * 로그인 처리 / Login Handler
 *
 * **비즈니스 규칙 / Business Rule**:
 * - 3회 실패 시 5분 잠금 (무차별 대입 공격 방지)
 * - JWT 토큰 발급 (만료: 7일)
 *
 * **Context**: 사용자가 "로그인" 버튼 클릭
 */
class LoginScreen extends StatefulWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: SafeArea(
        child: Padding(
          padding: EdgeInsets.all(24),
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              // 로고
              Image.asset('assets/logo.png', height: 80),
              SizedBox(height: 16),

              // 캐치프레이즈
              Text(
                '절대 안 늦는 습관 만들기',
                style: Theme.of(context).textTheme.titleMedium,
              ),
              SizedBox(height: 48),

              // 이메일 입력
              TextField(
                controller: _emailController,
                decoration: InputDecoration(
                  labelText: '이메일',
                  prefixIcon: Icon(Icons.email),
                ),
                keyboardType: TextInputType.emailAddress,
              ),
              SizedBox(height: 16),

              // 비밀번호 입력
              TextField(
                controller: _passwordController,
                decoration: InputDecoration(
                  labelText: '비밀번호',
                  prefixIcon: Icon(Icons.lock),
                  suffixIcon: IconButton(
                    icon: Icon(_obscurePassword ? Icons.visibility : Icons.visibility_off),
                    onPressed: () => setState(() => _obscurePassword = !_obscurePassword),
                  ),
                ),
                obscureText: _obscurePassword,
              ),

              // 로그인 버튼
              SizedBox(height: 32),
              ElevatedButton(
                onPressed: _handleLogin,
                child: Text('로그인'),
                style: ElevatedButton.styleFrom(
                  minimumSize: Size(double.infinity, 52),
                ),
              ),

              // 소셜 로그인
              SizedBox(height: 24),
              _buildSocialLoginButtons(),
            ],
          ),
        ),
      ),
    );
  }
}
```

```dart
/**
 * 대시보드 메인 위젯 / Dashboard Main Widget
 *
 * **Context**: 앱 실행 시 가장 먼저 보이는 화면
 * **비즈니스 규칙 / Business Rule**:
 * - 가장 임박한 일정을 자동으로 표시
 * - 실시간 카운트다운 (1초마다 업데이트)
 */
class DashboardScreen extends StatefulWidget {
  @override
  _DashboardScreenState createState() => _DashboardScreenState();
}

class _DashboardScreenState extends State<DashboardScreen> {
  Trip? _nextTrip;
  Timer? _countdownTimer;

  @override
  void initState() {
    super.initState();
    _loadNextTrip();
    _startCountdown();
  }

  void _startCountdown() {
    _countdownTimer = Timer.periodic(Duration(seconds: 1), (timer) {
      setState(() {
        // 카운트다운 갱신
      });
    });
  }

  Widget build(BuildContext context) {
    if (_nextTrip == null) {
      return _buildEmptyState();
    }

    final urgency = _calculateUrgency(_nextTrip!);

    return Scaffold(
      body: SafeArea(
        child: Column(
          children: [
            _buildHeader(),
            _buildCountdownCard(_nextTrip!, urgency),
            _buildRouteMap(_nextTrip!),
            _buildUpcomingSchedule(),
            _buildAddButton(),
          ],
        ),
      ),
    );
  }

  UrgencyLevel _calculateUrgency(Trip trip) {
    final now = DateTime.now();
    final timeUntilDeparture = trip.departureTime.difference(now);

    if (timeUntilDeparture.inMinutes <= 5) {
      return UrgencyLevel.critical;
    } else if (timeUntilDeparture.inMinutes <= 30) {
      return UrgencyLevel.warning;
    } else {
      return UrgencyLevel.normal;
    }
  }
}

enum UrgencyLevel {
  normal,   // 1시간 이상 여유
  warning,  // 30분 이내
  critical, // 10분 이내
}
```
